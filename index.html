function movePlayer(dx, dy) { if (!playerTurn) return; const nx = player.x + dx; const ny = player.y + dy; if (nx < 0 || nx >= BOARD_SIZE || ny < 0 || ny >= BOARD_SIZE) { log("Can't move off board."); return; } render(); // ensure player exists const startTile = getTile(player.x, player.y); const endTile = getTile(nx, ny); const playerDiv = startTile.querySelector(".player"); const nameTag = startTile.querySelector(".nametag"); if (!playerDiv || !nameTag) { // fallback player.x = nx; player.y = ny; if (player.shieldCooldown > 0) player.shieldCooldown--; log(playerName + " moves to " + nx + "," + ny); endPlayerTurn(); return; } // Get center positions of tiles const startRect = startTile.getBoundingClientRect(); const endRect = endTile.getBoundingClientRect(); const startX = startRect.left + startRect.width / 2; const startY = startRect.top + startRect.height / 2; const endX = endRect.left + endRect.width / 2; const endY = endRect.top + endRect.height / 2; const dxPixels = endX - startX; const dyPixels = endY - startY; // Animate real player playerDiv.style.transition = "transform 0.25s linear"; nameTag.style.transition = "transform 0.25s linear"; playerDiv.style.transform = `translate(${dxPixels}px, ${dyPixels}px)`; nameTag.style.transform = `translate(${dxPixels}px, ${dyPixels}px)`; setTimeout(() => { // Reset transform and move to new tile playerDiv.style.transition = ""; nameTag.style.transition = ""; playerDiv.style.transform = ""; nameTag.style.transform = ""; player.x = nx; player.y = ny; if (player.shieldCooldown > 0) player.shieldCooldown--; log(playerName + " moves to " + nx + "," + ny); render(); endPlayerTurn(); }, 250); }
